\begin{pseudo}*
\hd{Table-Insert}(T,x)\\
if $T.size \== 0$\\+
    \tn{allocate $T.table$ with 1 slot}\\
    $T.size \== 1$\\-
if $T.num \== T.size$\\+
    \tn{allocate \id{new-table} with $2\cdot T.size$ slots}\\
    \tn{insert all items in $T.table$ into \id{new-table}}\\
    \tn{free $T.table$}\\
    $T.table = \id{new-table}$\\
    $T.size = 2\cdot T.size$\\-
\tn{insert $x$ into $T.table$}\\
$T.num = T.num + 1$
\end{pseudo}
    \end{block}
